DROP TRIGGER trigger_demo;

CREATE TRIGGER trigger_demo AFTER UPDATE ON students
FOR EACH ROW
BEGIN
	INSERT INTO journal (type_operation, table_name, old_full_name, new_full_name)
	VALUES ('update', 'students', OLD.first_name + OLD.last_name, NEW.first_name + NEW.last_name);
END;
